class Solution {
    public List<String> summaryRanges(int[] nums) {

        if(nums.length == 0){
            return new ArrayList<>();
        }

        List<String> result = new ArrayList<>();
        int tmp = nums[0];
        boolean next;
        for (int i = 0; i <= nums.length - 1; ++i) {
            next = false;
            for (int j = 0; j + i + 1 < nums.length; j++) {
                if ((nums[i + j] + 1) == nums[i + j + 1]) {
                    continue;
                } else {
                    if (j == 0)
                        result.add(nums[i] + "");
                    else {
                        result.add(nums[i] + "->" + nums[i + j]);
                        i = i + j;
                    }
                    next = true;
                    break;
                }
            }

            if (!next) {
                if (i == nums.length - 1)
                    result.add(nums[nums.length - 1] + "");
                else {
                    //最后必须要退出循环
                    result.add(nums[i] + "->" + nums[nums.length - 1]);
                    break;
                }
            }
        }
        return result;
    }
}
