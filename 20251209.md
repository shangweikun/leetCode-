class Solution {
    public int[] plusOne(int[] digits) {
        boolean b = false;
        for(int i = digits.length-1; i >=0 ;i--){
            b = false;
            if(digits[i] == 9){
                digits[i] = 0;
                b = true;
            }else{
                digits[i]=digits[i]+1;
            }
            if(!b){
                break;
            }
        }
        int[] result;
        if(b){
            result = new int[digits.length+1];
            System.arraycopy(digits,0,result,1,digits.length);
            result[0] = 1;
        }else{
            result = digits;
        }
        return result;
    }
}

class Solution {
    public int[] plusOne(int[] digits) {
        int n = digits.length;
        for (int i = n - 1; i >= 0; i--) {
            if (digits[i] != 9) {
                ++digits[i];
                for (int j = i + 1; j < n; j++) {
                    digits[j] = 0;
                }
                return digits;
            }
        }
        int[] result = new int[n + 1];
        result[0] = 1;
        return result;
    }
}
