https://leetcode.cn/problems/string-compression/

```java
class Solution {
    public int compress(char[] chars) {

        if (chars.length == 1) {
            return 1;
        }
        int i = 0;
        for (int j = 0; i < chars.length && j < chars.length;) {
            for (; j < chars.length - 1; j++) {
                if (chars[j] != chars[j + 1]) {
                    break;
                }
            }
            if (i == j) {
                i++;
                j++;
            } else {
                int tmp = j - i + 1;
                for (; tmp / 10 > 0; tmp = tmp % 10) {
                    char c = (char) ((tmp / 10) + '0');
                    i++;
                    chars[i]=c;
                }
                char c = (char) ((tmp % 10) + '0');
                i++;
                chars[i]=c;
                
                i++;
                j++;
            }
        }
        return i;
    }
}
```
