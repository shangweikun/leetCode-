class Solution {
    public int compress(char[] chars) {

        if (chars.length == 1) {
            return 1;
        }
        int i = 0;
        for (int j = 0; i < chars.length && j < chars.length;) {
            int tmpJ = j;
            for (; j < chars.length - 1; j++) {
                if (chars[j] != chars[j + 1]) {
                    break;
                }
            }
            chars[i]=chars[j];
            if (j==tmpJ) {
                i++;
                j++;
            } else {
                int tmp = j - tmpJ + 1;
                char[] charArray = String.valueOf(i).toCharArray();
                for (char c1 : charArray) {
                    i++;
                    chars[i]=c1;
                }

                i++;
                j++;
            }
        }
        return i;
    }
}
